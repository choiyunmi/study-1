Q. for 문으로 5개의 input 을 만들되, 짝수는 text / 홀수는 checkbox 를 출력하세요.

Q. for 문으로 구구단 8단을 출력하세요.

Q. for/while 문 2개로 구구단을 출력하세요.

$(function(){
	// exam1 [s]
	var elExam1 = $('.exam_q').eq(0),
		i = 0,
		sResult1 = '';
	
	while (i < 5) {
		sResult1 += '
';
		i++;
	}

	elExam1.append(sResult1);
	// exam1 [e]


	// exam2 [s]
	var elExam2 = $('.exam_q').eq(1),
		sResult2 = '';

	for (var i = 0; i < 5; i++){
		if (i % 2 == 0) {
			sResult2 += '
';
		} else {
			sResult2 += '
';
		}
	}

	elExam2.append(sResult2);
	// exam2 [e]


	// exam3 [s]
	var elExam3 = $('.exam_q').eq(2),
		sResult3 = '';

	for (var i = 1; i < 10; i++){
		sResult3 += '8 X ' + i + ' = ' + 8 * i + ' 
';
	}

	elExam3.append(sResult3);
	// exam3 [e]
	

	// exam4 [s]
	var elExam4 = $('.exam_q').eq(3),
		i = 2,
		sResult4 ='';

	while (i < 10) {
		for (var j = 1; j < 10; j++) {
			sResult4 += i + ' X ' + j + ' = ' + i * j + '     ';
		}
		sResult4 += '
';
		i++;
	}

	elExam4.append(sResult4);
	// exam4 [e]
});
