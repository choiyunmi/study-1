타이머 꺼짐

$(function(){
	// exam1 [s]
	var elExam1Wrap = $('.exam_wrap').eq(0),
		elExam1Item = elExam1Wrap.find('.exam_q li'),
		elExam1Btn = elExam1Wrap.find('button'),
		nExam1Max = elExam1Item.length - 1,  // 최대 index값
		nExam1Current = -1;  // 현재 index값(첫번째 전)

	elExam1Btn.on('click', function () {
		nExam1Current++;  // index값 1 증가

		// 현재 index값이 최대 index값보다 크면
		if (nExam1Current > nExam1Max) {
			alert('end');
			nExam1Current = nExam1Max;  // index값 마지막 index로 변경
		}

		elExam1Item.eq(nExam1Current).css('background-color', 'yellow');
	});
	// exam1 [e]


	// exam2 [s]
	var elExam2Wrap = $('.exam_wrap').eq(1),
		elExam2Item = elExam2Wrap.find('.exam_q li'),
		elExam2Btn = elExam2Wrap.find('button'),
		nExam2Max = elExam2Item.length -1,  // 최대 index값
		nExam2Current = nExam2Max + 1;  // 현재 index값(마지막 다음)

	elExam2Btn.on('click', function () {
		nExam2Current--;  // index값 1 감소

		// 현재 index값이 0보다 작으면
		if (nExam2Current < 0) {
			alert('end');
			nExam2Current = 0;  // index값 첫번째 index로 변경
		}

		elExam2Item.eq(nExam2Current).css('background-color', 'yellow');
	});
	// exam2 [e]


	// exam3 [s]
	var elExam3Wrap = $('.exam_wrap').eq(2),
		elExam3Item = elExam3Wrap.find('.exam_q li'),
		elExam3PrevBtn = elExam3Wrap.find('.btn_prev'),
		elExam3NextBtn = elExam3Wrap.find('.btn_next'),
		nExam3Max = elExam3Item.length - 1,  // 최대 index값
		nExam3Current = 0;  // 현재 index값(0번)

	var changeExam3Active = function (direction) {
		nExam3Current = nExam3Current + direction;  // index값 +1 또는 -1

		if (nExam3Current > nExam3Max) {
			alert('end');
			nExam3Current = nExam3Max;  // index값 마지막 index로 변경
		} else if (nExam3Current < 0) {
			alert('end');
			nExam3Current = 0;  // index값 첫번째 index로 변경
		}

		elExam3Item.eq(nExam3Current).css('background-color', 'yellow').siblings().css('background-color','');
	};

	elExam3NextBtn.on('click', function () {
		changeExam3Active(1);
	});

	elExam3PrevBtn.on('click', function () {
		changeExam3Active(-1);
	});
	// exam3 [e]


	// exam4 [s]
	var elExam4Wrap = $('.exam_wrap').eq(3),
		elExam4Item = elExam4Wrap.find('.exam_q li'),
		elExam4PrevBtn = elExam4Wrap.find('.btn_prev'),
		elExam4NextBtn = elExam4Wrap.find('.btn_next'),
		nExam4Max = elExam4Item.length - 1,  // 최대 index값
		nExam4Current = 0;  // 현재 index값(0번)

	var changeExam4Active = function (direction) {
		nExam4Current = nExam4Current + direction;  // index값 +1 또는 -1

		if (nExam4Current > nExam4Max) {
			nExam4Current = 0;  // index값 첫번째 index로 변경
		} else if (nExam4Current < 0) {
			nExam4Current = nExam4Max;  // index값 마지막 index로 변경
		}

		elExam4Item.eq(nExam4Current).addClass('active').siblings().removeClass('active');
	};

	elExam4NextBtn.on('click', function () {
		changeExam4Active(1);
	});

	elExam4PrevBtn.on('click', function () {
		changeExam4Active(-1);
	});
	// exam4 [e]


	// exam5 [s]
	var elExam5Wrap = $('.exam_wrap').eq(4),
		elExam5Item = elExam5Wrap.find('.exam_q li'),
		elExam5Status = elExam5Wrap.find('.exam_q p'),
		elExam5PrevBtn = elExam5Wrap.find('.btn_prev'),
		elExam5NextBtn = elExam5Wrap.find('.btn_next'),
		elExam5PlayBtn = elExam5Wrap.find('.btn_play'),
		elExam5StopBtn = elExam5Wrap.find('.btn_stop'),
		nExam5Max = elExam5Item.length - 1,  // 최대 index값
		nExam5Current = 0,  // 현재 index값(0번)
		oTimer = null,
		bTimerOn = false,
		nTimerDelay = 1000;

	var move = function (direction) {
		nExam5Current = nExam5Current + direction;  // index값 +1 또는 -1

		if (nExam5Current > nExam5Max) {
			nExam5Current = 0;  // index값 첫번째 index로 변경
		} else if (nExam5Current < 0) {
			nExam5Current = nExam5Max;  // index값 마지막 index로 변경
		}

		elExam5Item.eq(nExam5Current).addClass('active').siblings().removeClass('active');
	};

	var play = function () {
		if (!bTimerOn) {
			bTimerOn = true;
			oTimer = setInterval(function () { move(1) }, nTimerDelay);
			elExam5Status.addClass('active').text('타이머 켜짐');
		}
	};

	var stop = function () {
		if (bTimerOn) {
			bTimerOn = false;
			clearInterval(oTimer);
			elExam5Status.removeClass('active').text('타이머 꺼짐');
		}
	};

	elExam5NextBtn.on('click', function () {
		stop();
		move(1);
	});

	elExam5PrevBtn.on('click', function () {
		stop();
		move(-1);
	});

	elExam5PlayBtn.on('click', play);
	
	elExam5StopBtn.on('click', stop);
	// exam5 [e]
});
