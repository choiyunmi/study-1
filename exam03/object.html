Q. 함수 호출시 new 를 붙인 것과 안 붙인 것의 차이를 설명하세요.

예시)
var obj = function () {
   console.log(this);
};
obj();  // window
new obj();  // obj

함수 호출시 new 연산자를 붙이면 함수 내 this는 생성된 객체 자신이다.
함수 호출시 new 연산자를 붙이지 않으면 함수 내 this는 global object(웹브라우저에서는 window)이다.

Q. 객체 내에서의 this 를 설명하세요.

예시)
var obj = {
   init : function () {
      console.log(this);
   }
};
obj.init();  // obj

객체 내에서 this는 해당 객체를 가리킨다.

var exam = exam || {};
exam.View = exam.View || {};

exam.View.exam6Object.prototype = {
	oDefault: {
		e : 'e',
		d : 'd',
		b : 'b',
		g : 'g',
		c : 'c',
		f : 'f',
		a : 'a'
	},
	init: function() {
		this._assignElements();
		this._attachEventHandlers();
	},
	_assignElements: function() {
		this._welExamWrap = $('.exam_wrap').eq(5);
		this._welExamResult = this._welExamWrap.find('p');
		this._welExamBtn = this._welExamWrap.find('button');

	_attachEventHandlers: function() {
		this._welExamBtn.on('click', $.proxy(this._onClickBtn, this));
	},

		var aTemp = [];

		for (var key in this.oDefault) {
			aTemp.push(this.oDefault[key]);
		}

		this._welExamResult.text(aTemp.sort().join('')).css('border-color', '#ff5b59');
	}
};
